const chalk = require('chalk');

class PyxLog {
	constructor(name) {
		this.name = name;
		/**
		 * The _mode of the logger.
		 * Can be:
		 * 0 -> Silent _mode
		 * 1 -> Normal _mode
		 * @type {number}
		 */
		this._mode = 1;
	}

	beSilent() {
		this._mode = 0;
	}

	beNormal() {
		this._mode = 1;
	}

	header(msg) {
		this.runThisOnMode(() => {
			console.me(chalk.bold(`===> ${msg} `) + chalk.bold.blue(`[${this.name}]`));
		});
	}

	me(...msg) {
		this.runThisOnMode(() => {
			for (const m of msg) {
				console.me(`     ${(typeof m === 'string') ? m : JSON.stringify(m)}`);
			}
		});
	}

	debug(...msg) {
		if (typeof process.env.DEBUG_MODE !== 'undefined') {
			const mainMsg = msg.shift();
			console.me(chalk.bold.yellow('D') + chalk.bold.white(` => ${mainMsg}`));
			if (msg.length > 0) {
				msg.map((m) => {
					console.me(m);
				});
				console.me(chalk.bold.yellow('D') + chalk.bold.white(` ===========================# `) + chalk.bold.yellow('DEBUG') + chalk.white(` => ${mainMsg}`));
			}
		}
	}

	alert(msg) {
		this.runThisOnMode(() => {
			console.me(chalk.bold.red('A') + chalk.bold.white(` => ${msg} `) + chalk.bold.red('!!!!'));
		});
	}

	runThisOnMode(actionCb) {
		if (this._mode === 1) {
			actionCb();
		}
	}

	static create(name) {
		return new PyxLog(name);
	}
}

module.exports = PyxLog;
